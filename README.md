## Install

```bash
npm install --save usehotkey
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'usehotkey'
import 'usehotkey/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
